187  polePlacement_ = {alpha, beta, gamma, lagFreq};

200  dcmPropGain_ = omega * (zmpPropGain - 1.);

253 void Stabilizer::reconfigure()
254 {
255  fdqpWeights_.configure(config_("fdqp_weights"));
256  if(config_.has("admittance"))
257  {
258  auto admittance = config_("admittance");
259  comAdmittance_ = admittance("com");
260  copAdmittance_ = admittance("cop");
261  dfzAdmittance_ = admittance("dfz");
262  dfzDamping_ = admittance("dfz_damping");
263  }
264  if(config_.has("dcm_tracking"))
265  {
266  auto dcmTracking = config_("dcm_tracking");
267  dcmPropGain_ = dcmTracking("gains")("prop");
268  dcmIntegralGain_ = dcmTracking("gains")("integral");
269  dcmDerivGain_ = dcmTracking("gains")("deriv");
270  dcmDerivator_.timeConstant(dcmTracking("derivator_time_constant"));
271  dcmIntegrator_.timeConstant(dcmTracking("integrator_time_constant"));
272  }
273  if(config_.has("tasks"))
274  {
275  auto tasks = config_("tasks");
276  if(tasks.has("com"))
277  {
278  tasks("com")("active_joints", comActiveJoints_);
279  tasks("com")("stiffness", comStiffness_);
280  tasks("com")("weight", comWeight_);
281  }
282  if(tasks.has("contact"))
283  {
284  double d = tasks("contact")("damping");
285  double k = tasks("contact")("stiffness");
286  contactDamping_ = sva::MotionVecd({d, d, d}, {d, d, d});
287  contactStiffness_ = sva::MotionVecd({k, k, k}, {k, k, k});
288  tasks("contact")("stiffness", contactStiffness_);
289  tasks("contact")("weight", contactWeight_);
290  }
291  if(tasks.has("swing_foot"))
292  {
293  tasks("swing_foot")("stiffness", swingFootStiffness_);
294  tasks("swing_foot")("weight", swingFootWeight_);
295  }
296  }
297  if(config_.has("vdc"))
298  {
299  auto vdc = config_("vdc");
300  vdcFrequency_ = vdc("frequency");
301  vdcStiffness_ = vdc("stiffness");
302  }
303  if(config_.has("zmpcc"))
304  {
305  auto zmpcc = config_("zmpcc");
306  zmpccIntegrator_.rate(zmpcc("integrator_leak_rate"));
307  }
308 }

310 void Stabilizer::reset(const mc_rbdyn::Robots & robots)
311 {
312  unsigned robotIndex = robots.robotIndex();
313 
314  comTask.reset(new mc_tasks::CoMTask(robots, robotIndex));
315  comTask->selectActiveJoints(comActiveJoints_);
316  comTask->setGains(comStiffness_, 2 * comStiffness_.cwiseSqrt());
317  comTask->weight(comWeight_);
318 
319  leftFootTask.reset(new mc_tasks::force::CoPTask("LeftFootCenter", robots, robotIndex));
320  rightFootTask.reset(new mc_tasks::force::CoPTask("RightFootCenter", robots, robotIndex));
321  leftFootTask->maxAngularVel({MAX_FDC_RX_VEL, MAX_FDC_RY_VEL, MAX_FDC_RZ_VEL});
322  rightFootTask->maxAngularVel({MAX_FDC_RX_VEL, MAX_FDC_RY_VEL, MAX_FDC_RZ_VEL});
323  setContact(leftFootTask, leftFootTask->surfacePose());
324  setContact(rightFootTask, rightFootTask->surfacePose());
325 
326  dcmDerivator_.setZero();
327  dcmIntegrator_.saturation(MAX_AVERAGE_DCM_ERROR);
328  dcmIntegrator_.setZero();
329  zmpccIntegrator_.saturation(MAX_ZMPCC_COM_OFFSET);
330  zmpccIntegrator_.setZero();
331 
332  Eigen::Vector3d staticForce = -mass_ * world::gravity;
333 
334  dcmAverageError_ = Eigen::Vector3d::Zero();
335  dcmError_ = Eigen::Vector3d::Zero();
336  dcmVelError_ = Eigen::Vector3d::Zero();
337  dfzForceError_ = 0.;
338  dfzHeightError_ = 0.;
339  distribWrench_ = {pendulum_.com().cross(staticForce), staticForce};
340  vdcHeightError_ = 0.;
341  zmpError_ = Eigen::Vector3d::Zero();
342  zmpccCoMAccel_ = Eigen::Vector3d::Zero();
343  zmpccCoMOffset_ = Eigen::Vector3d::Zero();
344  zmpccCoMVel_ = Eigen::Vector3d::Zero();
345  zmpccError_ = Eigen::Vector3d::Zero();
346 }

402 bool Stabilizer::detectTouchdown(const std::shared_ptr<mc_tasks::force::CoPTask> footTask, const Contact & contact)
403 {
404  const sva::PTransformd X_0_s = footTask->surfacePose();
405  const sva::PTransformd & X_0_c = contact.pose;
406  sva::PTransformd X_c_s = X_0_s * X_0_c.inv();
407  double xDist = std::abs(X_c_s.translation().x());
408  double yDist = std::abs(X_c_s.translation().y());
409  double zDist = std::abs(X_c_s.translation().z());
410  double Fz = footTask->measuredWrench().force().z();
411  return (xDist < 0.03 && yDist < 0.03 && zDist < 0.03 && Fz > 50.);
412 }

414 void Stabilizer::seekTouchdown(std::shared_ptr<mc_tasks::force::CoPTask> footTask)
415 {
416  constexpr double MAX_VEL = 0.01; // [m] / [s]
417  constexpr double TOUCHDOWN_FORCE = 50.; // [N]
418  constexpr double DESIRED_AFZ = MAX_VEL / TOUCHDOWN_FORCE;
419  if(footTask->measuredWrench().force().z() < TOUCHDOWN_FORCE)
420  {
421  auto a = footTask->admittance();
422  double AFz = clamp(DESIRED_AFZ, 0., 1e-2, "Contact seeking admittance");
423  footTask->admittance({a.couple(), {a.force().x(), a.force().y(), AFz}});
424  footTask->targetForce({0., 0., TOUCHDOWN_FORCE});
425  }
426 }

428 void Stabilizer::setSupportFootGains()
429 {
430  sva::MotionVecd vdcContactStiffness = {contactStiffness_.angular(), {vdcStiffness_, vdcStiffness_, vdcStiffness_}};

432  {
433  case ContactState::DoubleSupport:
434  leftFootTask->admittance(contactAdmittance());

436  rightFootTask->admittance(contactAdmittance());

442  break;
443  case ContactState::RightFoot:
444  rightFootTask->admittance(contactAdmittance());
445  rightFootTask->setGains(vdcContactStiffness, contactDamping_);
446  break;
447  }
448 }

457 void Stabilizer::updateZMPFrame()
458 {
459  const sva::PTransformd & X_0_lc = leftFootContact.pose;
460  const sva::PTransformd & X_0_rc = rightFootContact.pose;
461  switch(contactState_)
462  {
463  case ContactState::DoubleSupport:
464  zmpFrame_ = sva::interpolate(X_0_lc, X_0_rc, 0.5);
465  break;
466  case ContactState::LeftFoot:
467  zmpFrame_ = X_0_lc;
468  break;
469  case ContactState::RightFoot:
470  zmpFrame_ = X_0_rc;
471  break;
472  }
473  measuredZMP_ = computeZMP(measuredWrench_);
474 }

476 Eigen::Vector3d Stabilizer::computeZMP(const sva::ForceVecd & wrench) const
477 {
478  Eigen::Vector3d n = zmpFrame_.rotation().row(2);
479  Eigen::Vector3d p = zmpFrame_.translation();
480  const Eigen::Vector3d & force = wrench.force();
481  double normalForce = n.dot(force);
482  if(normalForce < 1.)
483  {
484  double lambda = std::pow(pendulum_.omega(), 2);
485  return measuredCoM_ + world::gravity / lambda; // default for logging
486  }
487  const Eigen::Vector3d & moment_0 = wrench.couple();
488  Eigen::Vector3d moment_p = moment_0 - p.cross(force);
489  return p + n.cross(moment_p) / normalForce;
490 }

492 void Stabilizer::run()
493 {
494  auto startTime = std::chrono::high_resolution_clock::now();
495  checkGains();
496  checkInTheAir();
497  setSupportFootGains();
498  updateZMPFrame();
499  auto desiredWrench = computeDesiredWrench();
500  switch(contactState_)
501  {
502  case ContactState::DoubleSupport:
503  distributeWrench(desiredWrench);
504  break;
505  case ContactState::LeftFoot:
506  saturateWrench(desiredWrench, leftFootTask);
507  rightFootTask->setZeroTargetWrench();
508  break;
509  case ContactState::RightFoot:
510  saturateWrench(desiredWrench, rightFootTask);
511  leftFootTask->setZeroTargetWrench();
512  break;
513  }
514  updateCoMTaskZMPCC();
515  updateFootForceDifferenceControl();
516  auto endTime = std::chrono::high_resolution_clock::now();
517  runTime_ = std::chrono::duration<double, std::milli>(endTime - startTime).count();
518 }

531 sva::ForceVecd Stabilizer::computeDesiredWrench()
532 {
533  double omega = pendulum_.omega();
534  Eigen::Vector3d comError = pendulum_.com() - measuredCoM_;
535  Eigen::Vector3d comdError = pendulum_.comd() - measuredCoMd_;
536  dcmError_ = comError + comdError / omega;
537  dcmError_.z() = 0.;
538 
539  if(inTheAir_)
540  {
541  dcmDerivator_.setZero();
542  dcmIntegrator_.append(Eigen::Vector3d::Zero());
543  }
544  else
545  {
546  zmpError_ = pendulum_.zmp() - measuredZMP_; // XXX: both in same plane?
547  zmpError_.z() = 0.;
548  dcmDerivator_.update(omega * (dcmError_ - zmpError_));
549  dcmIntegrator_.append(dcmError_);
550  }
551  dcmAverageError_ = dcmIntegrator_.eval();
552  dcmVelError_ = dcmDerivator_.eval();
553 
554  Eigen::Vector3d desiredCoMAccel = pendulum_.comdd();
555  desiredCoMAccel += omega * (dcmPropGain_ * dcmError_ + comdError);
556  desiredCoMAccel += omega * dcmIntegralGain_ * dcmAverageError_;
557  desiredCoMAccel += omega * dcmDerivGain_ * dcmVelError_;
558  auto desiredForce = mass_ * (desiredCoMAccel - world::gravity);
559 
560  // Previous implementation (up to v1.3):
561  // return {pendulum_.com().cross(desiredForce), desiredForce};
562  // See https://github.com/stephane-caron/lipm_walking_controller/issues/28
563 
564  return {measuredCoM_.cross(desiredForce), desiredForce};
565 }

567 void Stabilizer::distributeWrench(const sva::ForceVecd & desiredWrench)
568 {
569  // Variables
570  // ---------
571  // x = [w_l_0 w_r_0] where
572  // w_l_0: spatial force vector of left foot contact in inertial frame
573  // w_r_0: spatial force vector of right foot contact in inertial frame
574  //
575  // Objective
576  // ---------
577  // Weighted minimization of the following tasks:
578  // w_l_0 + w_r_0 == desiredWrench -- realize desired contact wrench
579  // w_l_lankle == 0 -- minimize left foot ankle torque (anisotropic weight)
580  // w_r_rankle == 0 -- minimize right foot ankle torque (anisotropic weight)
581  // (1 - lfr) * w_l_lc.z() == lfr * w_r_rc.z()
582  //
583  // Constraints
584  // -----------
585  // CWC X_0_lc* w_l_0 <= 0 -- left foot wrench within contact wrench cone
586  // CWC X_0_rc* w_r_0 <= 0 -- right foot wrench within contact wrench cone
587  // (X_0_lc* w_l_0).z() > MIN_DSP_FZ -- minimum left foot contact force
588  // (X_0_rc* w_r_0).z() > MIN_DSP_FZ -- minimum right foot contact force
589 
590  const sva::PTransformd & X_0_lc = leftFootContact.pose;
591  const sva::PTransformd & X_0_rc = rightFootContact.pose;
592  sva::PTransformd X_0_lankle = leftFootContact.anklePose(sole_);
593  sva::PTransformd X_0_rankle = rightFootContact.anklePose(sole_);
594 
595  constexpr unsigned NB_VAR = 6 + 6;
596  constexpr unsigned COST_DIM = 6 + NB_VAR + 1;
597  Eigen::MatrixXd A;
598  Eigen::VectorXd b;
599  A.setZero(COST_DIM, NB_VAR);
600  b.setZero(COST_DIM);
601 
602  // |w_l_0 + w_r_0 - desiredWrench|^2
603  auto A_net = A.block<6, 12>(0, 0);
604  auto b_net = b.segment<6>(0);
605  A_net.block<6, 6>(0, 0) = Eigen::Matrix6d::Identity();
606  A_net.block<6, 6>(0, 6) = Eigen::Matrix6d::Identity();
607  b_net = desiredWrench.vector();
608 
609  // |ankle torques|^2
610  auto A_lankle = A.block<6, 6>(6, 0);
611  auto A_rankle = A.block<6, 6>(12, 6);
612  // anisotropic weights: taux, tauy, tauz, fx, fy, fz;
613  A_lankle.diagonal() << 1., 1., 1e-4, 1e-3, 1e-3, 1e-4;
614  A_rankle.diagonal() << 1., 1., 1e-4, 1e-3, 1e-3, 1e-4;
615  A_lankle *= X_0_lankle.dualMatrix();
616  A_rankle *= X_0_rankle.dualMatrix();
617 
618  // |(1 - lfr) * w_l_lc.force().z() - lfr * w_r_rc.force().z()|^2
619  double lfr = leftFootRatio_;
620  auto A_fratio = A.block<1, 12>(18, 0);
621  A_fratio.block<1, 6>(0, 0) = (1 - lfr) * X_0_lc.dualMatrix().bottomRows<1>();
622  A_fratio.block<1, 6>(0, 6) = -lfr * X_0_rc.dualMatrix().bottomRows<1>();
623 
624  // Apply weights
625  A_net *= fdqpWeights_.netWrenchSqrt;
626  b_net *= fdqpWeights_.netWrenchSqrt;
627  A_lankle *= fdqpWeights_.ankleTorqueSqrt;
628  A_rankle *= fdqpWeights_.ankleTorqueSqrt;
629  // b_lankle = 0
630  // b_rankle = 0
631  A_fratio *= fdqpWeights_.forceRatioSqrt;
632  // b_fratio = 0
633 
634  Eigen::MatrixXd Q = A.transpose() * A;
635  Eigen::VectorXd c = -A.transpose() * b;
636 
637  constexpr unsigned NB_CONS = 16 + 16 + 2;
638  Eigen::Matrix<double, NB_CONS, NB_VAR> A_ineq;
639  Eigen::VectorXd b_ineq;
640  A_ineq.setZero(NB_CONS, NB_VAR);
641  b_ineq.setZero(NB_CONS);
642  // CWC * w_l_lc <= 0
643  A_ineq.block<16, 6>(0, 0) = wrenchFaceMatrix_ * X_0_lc.dualMatrix();
644  // b_ineq.segment<16>(0) is already zero
645  // CWC * w_r_rc <= 0
646  A_ineq.block<16, 6>(16, 6) = wrenchFaceMatrix_ * X_0_rc.dualMatrix();
647  // b_ineq.segment<16>(16) is already zero
648  // w_l_lc.force().z() >= MIN_DSP_FZ
649  A_ineq.block<1, 6>(32, 0) = -X_0_lc.dualMatrix().bottomRows<1>();
650  b_ineq(32) = -MIN_DSP_FZ;
651  // w_r_rc.force().z() >= MIN_DSP_FZ
652  A_ineq.block<1, 6>(33, 6) = -X_0_rc.dualMatrix().bottomRows<1>();
653  b_ineq(33) = -MIN_DSP_FZ;
654 
655  qpSolver_.problem(NB_VAR, 0, NB_CONS);
656  Eigen::MatrixXd A_eq(0, 0);
657  Eigen::VectorXd b_eq;
658  b_eq.resize(0);
659 
660  bool solutionFound = qpSolver_.solve(Q, c, A_eq, b_eq, A_ineq, b_ineq, /* isDecomp = */ false);
661  if(!solutionFound)
662  {
663  LOG_ERROR("DS force distribution QP: solver found no solution");
664  return;
665  }
666 
667  Eigen::VectorXd x = qpSolver_.result();
668  sva::ForceVecd w_l_0(x.segment<3>(0), x.segment<3>(3));
669  sva::ForceVecd w_r_0(x.segment<3>(6), x.segment<3>(9));
670  distribWrench_ = w_l_0 + w_r_0;
671 
672  sva::ForceVecd w_l_lc = X_0_lc.dualMul(w_l_0);
673  sva::ForceVecd w_r_rc = X_0_rc.dualMul(w_r_0);
674  Eigen::Vector2d leftCoP = (e_z.cross(w_l_lc.couple()) / w_l_lc.force()(2)).head<2>();
675  Eigen::Vector2d rightCoP = (e_z.cross(w_r_rc.couple()) / w_r_rc.force()(2)).head<2>();
676  leftFootTask->targetCoP(leftCoP);
677  leftFootTask->targetForce(w_l_lc.force());
678  rightFootTask->targetCoP(rightCoP);
679  rightFootTask->targetForce(w_r_rc.force());
680 }

682 void Stabilizer::saturateWrench(const sva::ForceVecd & desiredWrench,
683  std::shared_ptr<mc_tasks::force::CoPTask> & footTask)
684 {
685  constexpr unsigned NB_CONS = 16;
686  constexpr unsigned NB_VAR = 6;
687 
688  // Variables
689  // ---------
690  // x = [w_0] where
691  // w_0: spatial force vector of foot contact in inertial frame
692  //
693  // Objective
694  // ---------
695  // weighted minimization of |w_c - X_0_c* desiredWrench|^2
696  //
697  // Constraints
698  // -----------
699  // F X_0_c* w_0 <= 0 -- contact stability
700 
701  const sva::PTransformd & X_0_c = footTask->targetPose();
702 
703  Eigen::Matrix6d Q = Eigen::Matrix6d::Identity();
704  Eigen::Vector6d c = -desiredWrench.vector();
705 
706  Eigen::MatrixXd A_ineq = wrenchFaceMatrix_ * X_0_c.dualMatrix();
707  Eigen::VectorXd b_ineq;
708  b_ineq.setZero(NB_CONS);
709 
710  qpSolver_.problem(NB_VAR, 0, NB_CONS);
711  Eigen::MatrixXd A_eq(0, 0);
712  Eigen::VectorXd b_eq;
713  b_eq.resize(0);
714 
715  bool solutionFound = qpSolver_.solve(Q, c, A_eq, b_eq, A_ineq, b_ineq, /* isDecomp = */ true);
716  if(!solutionFound)
717  {
718  LOG_ERROR("SS force distribution QP: solver found no solution");
719  return;
720  }
721 
722  Eigen::VectorXd x = qpSolver_.result();
723  sva::ForceVecd w_0(x.head<3>(), x.tail<3>());
724  sva::ForceVecd w_c = X_0_c.dualMul(w_0);
725  Eigen::Vector2d cop = (e_z.cross(w_c.couple()) / w_c.force()(2)).head<2>();
726  footTask->targetCoP(cop);
727  footTask->targetForce(w_c.force());
728  distribWrench_ = w_0;
729 }

731 void Stabilizer::updateCoMTaskZMPCC()
732 {
733  if(zmpccOnlyDS_ && contactState_ != ContactState::DoubleSupport)
734  {
735  zmpccCoMAccel_.setZero();
736  zmpccCoMVel_.setZero();
737  zmpccIntegrator_.add(Eigen::Vector3d::Zero(), dt_); // leak to zero
738  }
739  else
740  {
741  auto distribZMP = computeZMP(distribWrench_);
742  zmpccError_ = distribZMP - measuredZMP_;
743  const Eigen::Matrix3d & R_0_c = zmpFrame_.rotation();
744  const Eigen::Transpose<const Eigen::Matrix3d> R_c_0 = R_0_c.transpose();
745  Eigen::Vector3d comAdmittance = {comAdmittance_.x(), comAdmittance_.y(), 0.};
746  Eigen::Vector3d newVel = -R_c_0 * comAdmittance.cwiseProduct(R_0_c * zmpccError_);
747  Eigen::Vector3d newAccel = (newVel - zmpccCoMVel_) / dt_;
748  zmpccIntegrator_.add(newVel, dt_);
749  zmpccCoMAccel_ = newAccel;
750  zmpccCoMVel_ = newVel;
751  }
752  zmpccCoMOffset_ = zmpccIntegrator_.eval();
753  comTask->com(pendulum_.com() + zmpccCoMOffset_);
754  comTask->refVel(pendulum_.comd() + zmpccCoMVel_);
755  comTask->refAccel(pendulum_.comdd() + zmpccCoMAccel_);
756 }

758 void Stabilizer::updateFootForceDifferenceControl()
759 {
760  if(contactState_ != ContactState::DoubleSupport || inTheAir_)
761  {
762  dfzForceError_ = 0.;
763  dfzHeightError_ = 0.;
764  vdcHeightError_ = 0.;
765  leftFootTask->refVelB({{0., 0., 0.}, {0., 0., 0.}});
766  rightFootTask->refVelB({{0., 0., 0.}, {0., 0., 0.}});
767  return;
768  }
769 
770  double LFz_d = leftFootTask->targetWrench().force().z();
771  double RFz_d = rightFootTask->targetWrench().force().z();
772  double LFz = leftFootTask->measuredWrench().force().z();
773  double RFz = rightFootTask->measuredWrench().force().z();
774  dfzForceError_ = (LFz_d - RFz_d) - (LFz - RFz);
775 
776  double LTz_d = leftFootTask->targetPose().translation().z();
777  double RTz_d = rightFootTask->targetPose().translation().z();
778  double LTz = leftFootTask->surfacePose().translation().z();
779  double RTz = rightFootTask->surfacePose().translation().z();
780  dfzHeightError_ = (LTz_d - RTz_d) - (LTz - RTz);
781  vdcHeightError_ = (LTz_d + RTz_d) - (LTz + RTz);
782 
783  double dz_ctrl = dfzAdmittance_ * dfzForceError_ - dfzDamping_ * dfzHeightError_;
784  double dz_vdc = vdcFrequency_ * vdcHeightError_;
785  sva::MotionVecd velF = {{0., 0., 0.}, {0., 0., dz_ctrl}};
786  sva::MotionVecd velT = {{0., 0., 0.}, {0., 0., dz_vdc}};
787  leftFootTask->refVelB(0.5 * (velT - velF));
788  rightFootTask->refVelB(0.5 * (velT + velF));
789 }
