218  double measuredLeftFootRatio()
219  {
220  double leftFootPressure = realRobot().forceSensor("LeftFootForceSensor").force().z();
221  double rightFootPressure = realRobot().forceSensor("RightFootForceSensor").force().z();
222  leftFootPressure = std::max(0., leftFootPressure);
223  rightFootPressure = std::max(0., rightFootPressure);
224  return leftFootPressure / (leftFootPressure + rightFootPressure);
